{
// 以下保留兼容旧代码的方法，但实际使用新的语言管理器
procedure TForm1.InitLanguageStrings;
begin
  // 只用于兼容旧的接口
  SwitchToLanguageCode(LanguageManager.CurrentLanguage);
end;
}

{
// 根据枚举设置语言
procedure TForm1.SetLanguage(Language: TAppLanguage);
begin
  // 转换为语言代码
  SwitchToLanguageCode(GetLanguageCodeByEnum(Language));
end;
}
